📝 判断逻辑说明

isMarkdownPath(path) {
    if (!path || path === '/' || path === '') return false;

    return path.startsWith('/') &&
           path.endsWith('/') &&
           path.length > 2 &&
           !path.startsWith('/lore/') &&
           !path.includes('.');
}
